    }

    // --- aplica estilo mínimo inline caso CSS externo esteja bloqueando ---
    (function ensureStyles(el){
      try{
        el.style.display = el.style.display || 'flex';
        el.style.alignItems = el.style.alignItems || 'center';
        el.style.gap = el.style.gap || '10px';
        el.style.padding = el.style.padding || '12px 16px';
        el.style.borderRadius = el.style.borderRadius || '10px';
        el.style.background = el.style.background || 'rgb(19,163,87)'; // fallback
        el.style.color = el.style.color || '#fff';
        el.style.zIndex = '2147483647';
        el.style.maxWidth = el.style.maxWidth || '360px';
        el.style.whiteSpace = 'normal';
        el.style.wordBreak = 'break-word';
      }catch(e){
        console.warn('[PAX] Falha ao ajustar estilos inline', e);
      }
    })(purchaseEl);

    // helpers
    function showText(text){
      purchaseEl.textContent = text;
      purchaseEl.classList.add('show');
    }
    function hidePop(){
      purchaseEl.classList.remove('show');
    }

    // força visibilidade (usa !important quando possível)
    function forceVisible(el){
      try{
        el.style.setProperty('display','flex','important');
        el.style.setProperty('visibility','visible','important');
      }catch(e){}
    }

    // lógica de exibição
    const VISIBLE_MS = 3500; // tempo visível
    let index = 0;

    function triggerPurchase(){
      const name = buyers[index % buyers.length];
      index++;
      const msg = ${name} — Acabou de comprar o PAX.;
      forceVisible(purchaseEl);
      showText(msg);
      setTimeout(hidePop, VISIBLE_MS);
      console.debug('[PAX] pop exibido:', msg);
    }

    // scheduling: primeiro depois de 10s, depois interval (45s)
    const FIRST_DELAY = 10000;
    const INTERVAL_MS = 45000;
    let intervalHandle = null;
    let firstTimeout = setTimeout(()=>{
      try{
        triggerPurchase();
        // programa o intervalo regular
        intervalHandle = setInterval(triggerPurchase, INTERVAL_MS);
      }catch(e){
        console.error('[PAX] erro no trigger inicial', e);
      }
    }, FIRST_DELAY);

    // expose API for debugging / manual trigger
    window.__PAX_POPOUTS = {
      triggerNow: () => { try{ triggerPurchase(); }catch(e){console.error(e);} },
      stop: () => { clearTimeout(firstTimeout); if(intervalHandle) clearInterval(intervalHandle); },
      _el: purchaseEl
    };

    console.info('[PAX] init ok — primeiro pop em', FIRST_DELAY/1000, 's e intervalo de', INTERVAL_MS/1000, 's');
  }

  if(document.readyState === 'loading'){
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }

})();
